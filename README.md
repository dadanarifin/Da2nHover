Da2nHover
=========

Hover Image Effect

=== Add this Script ===
jQuery(function(){
  jQuery('.NameClassImages').da2nhover();
});




=== Options ===

'startEffect'   : 'transparent', 

'hoverEffect'   : 'normal', //hovereffect: normal,sliceDown,sliceDownLeft,sliceUp,sliceUpLeft 

'imageOpacity'  : 0.5, 

'animSpeed'     : 300, 

'fillColor'     : '#000', 

'textColor'     : '#fff', 

'overlayText'   : '', 

'slices'        : 10, 

'boxCols'       : 5, 

'boxRows'       : 3, 

'MarginOut'     : 10, 

'ShadowOut'     : 10
